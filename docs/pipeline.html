🔧 Feature Engineering

# Các cột mới tạo ở Silver layer trip_id = MD5(VendorID|pickup|dropoff|PU|DO|pax|dist|fare|tip|total|pay) pickup_date = to_date(tpep_pickup_datetime) trip_duration_min = (dropoff - pickup) / 60 avg_speed_mph = trip_distance * 60 / trip_duration_min fare_per_mile = total_amount / trip_distance fare_per_min = total_amount / trip_duration_min is_tip_more_total = tip_amount > total_amount # Rename cột chuẩn hoá VendorID → vendor_id RatecodeID → ratecode_id PULocationID → pulocation_id DOLocationID → dolocation_id Airport_fee → airport_fee
